#include <stdint.h>
#include <stdio.h>
int main() {
   uint32_t input = 12005;
   uint8_t output[4];
   output[0] = (uint8_t)(input & 0xFF);         // LSB
   output[1] = (uint8_t)((input >> 8) & 0xFF);
   output[2] = (uint8_t)((input >> 16) & 0xFF);
   output[3] = (uint8_t)((input >> 24) & 0xFF);  // MSB
   printf("Bytes: {%d, %d, %d, %d}\n", output[0], output[1], output[2], output[3]);
   return 0;
}
